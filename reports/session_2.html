+def _apply_drawdown_cap(weights: pd.DataFrame, max_dd_pct: float) -> pd.DataFrame:
+    """Apply hard drawdown cap to portfolio: if cumulative returns exceed threshold,
+    force long positions down and shorts up proportionally."""
+    if max_dd_pct <= 0 or weights.sum().abs() == 0:
+        return weights
+
+    cumret = (1 + weights).cumprod(axis=0)
+    peak = cumret.cummax(axis=0)
+    dd = peak - cumret
+
+    mask = dd > max_dd_pct * np.ones_like(dd)
+    if not mask.any().any():
+        return weights
+
+    # Force positions to bring DD back under threshold, preserving sign structure
+    for date in mask.index:
+        w = weights.loc[date]
+        peak_val = cumret.loc[date].max()
+        target_dd = max_dd_pct * np.ones_like(w)
+        while dd.loc[date].sum() > target_dd.sum():
+            # Reduce longs, increase shorts (keep signs)
+            total_pos = w[w >= 0].sum()
+            if total_pos == 0:
+                break
+            reduction = min(1.0, (dd.loc[date][w >= 0].max() - target_dd[w >= 0]) / total_pos)
+            w[w >= 0] *= (1 - reduction)
+            dd.loc[date][w >= 0] -= reduction * np.abs(w[w >= 0])
+
+    weights.loc[mask, :] = w
+    return weights
